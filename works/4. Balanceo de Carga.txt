Servidor 1

server {
  listen 7101; # servidor oculto 1

  location /info {
    internal;
    rewrite /info /info.shtml?sid=1;
  }

  location ~ \.shtml$ {
    internal;
    ssi on;
    root /home/ubuntu/www/server;
  }

}
